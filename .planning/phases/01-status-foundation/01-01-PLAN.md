```typescript
export const EVENT_STATUS_COLORS = {
    PIPELINE: { bg, text, border, className, markerColor },
    COMMITTED: { bg, text, border, className, markerColor },
    OCCURRED: { bg, text, border, className, markerColor },
    CANCELED: { bg, text, border, className, markerColor }
} as const;

export type EventStatus = keyof typeof EVENT_STATUS_COLORS;

export const getStatusColor = (status: string) => {
    const key = status.toUpperCase();
    if (key in EVENT_STATUS_COLORS) {
        return EVENT_STATUS_COLORS[key as EventStatus];
    }
    return EVENT_STATUS_COLORS.PIPELINE;
};
```
